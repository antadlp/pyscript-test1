Simple Catenary

import matplotlib.pyplot as plt import numpy as np def catenary(**kwargs): a= kwargs.get("a", 1) x = kwargs.get("x", np.linspace(0, 1, 100)) c = kwargs.get("c", 0) return np.divide(np.cosh(a*x), a) + c fig, ax = plt.subplots() a = 1 x = np.linspace(-1, 1, 100) y = catenary(x=x, a=a, c=0) plt.plot(x, y, label=str(a)) plt.legend() fig
